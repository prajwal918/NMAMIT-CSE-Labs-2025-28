\section*{Experiment 1}
\addcontentsline{toc}{section}{Experiment 1}
\noindent\fcolorbox{linegray}{qbg}{%
\begin{minipage}{0.97\linewidth}
\textbf{Question:} Write an ALP to search a key element in a list of n 8-bit numbers using binary search.
\end{minipage}%
}
\vspace{0.4em}
\noindent\textbf{Answer}
\begin{lstlisting}[style=asmbase]
DATA SEGMENT
ARR DB 10H,20H,30H,40H,50H
LEN EQU ($-ARR)
KEY DB 40H
MSG1 DB 0DH,0AH,'KEY FOUND$'
MSG2 DB 0DH,0AH,'KEY NOT FOUND$'
DATA ENDS

CODE SEGMENT
ASSUME CS:CODE,DS:DATA
START:
    MOV AX,DATA
    MOV DS,AX

    MOV SI,0
    MOV DI,LEN
    DEC DI

BS_LOOP:
    CMP SI,DI
    JA NOT_FOUND

    MOV AX,SI
    ADD AX,DI
    SHR AX,1
    MOV BX,AX

    MOV AL,ARR[BX]
    CMP AL,KEY
    JE FOUND
    JB GO_RIGHT

    DEC BX
    MOV DI,BX
    JMP BS_LOOP

GO_RIGHT:
    INC BX
    MOV SI,BX
    JMP BS_LOOP

FOUND:
    MOV AH,09H
    LEA DX,MSG1
    INT 21H
    JMP EXIT

NOT_FOUND:
    MOV AH,09H
    LEA DX,MSG2
    INT 21H

EXIT:
    MOV AH,4CH
    INT 21H
CODE ENDS
END START
\end{lstlisting}

\section*{Experiment 2}
\addcontentsline{toc}{section}{Experiment 2}
\noindent\fcolorbox{linegray}{qbg}{%
\begin{minipage}{0.97\linewidth}
\textbf{Question:} Write ALP macros to read a character, display a character, then read and print a string terminated by carriage return.
\end{minipage}%
}
\vspace{0.4em}
\noindent\textbf{Answer}
\begin{lstlisting}[style=asmbase]
DATA SEGMENT
BUF DB 50 DUP(?)
MSG1 DB 0DH,0AH,'ENTER STRING:$'
MSG2 DB 0DH,0AH,'YOU ENTERED:$'
DATA ENDS

GETCHAR MACRO
    MOV AH,01H
    INT 21H
ENDM

PUTCHAR MACRO R
    MOV DL,R
    MOV AH,02H
    INT 21H
ENDM

CODE SEGMENT
ASSUME CS:CODE,DS:DATA
START:
    MOV AX,DATA
    MOV DS,AX

    MOV AH,09H
    LEA DX,MSG1
    INT 21H

    LEA SI,BUF
READ_LOOP:
    GETCHAR
    CMP AL,0DH
    JE PRINT_IT
    MOV [SI],AL
    INC SI
    JMP READ_LOOP

PRINT_IT:
    MOV BYTE PTR [SI],'$'

    MOV AH,09H
    LEA DX,MSG2
    INT 21H

    LEA SI,BUF
DISP_LOOP:
    MOV AL,[SI]
    CMP AL,'$'
    JE EXIT
    PUTCHAR AL
    INC SI
    JMP DISP_LOOP

EXIT:
    MOV AH,4CH
    INT 21H
CODE ENDS
END START
\end{lstlisting}

\section*{Experiment 3}

\noindent\fcolorbox{linegray}{qbg}{%
\begin{minipage}{0.97\linewidth}
\textbf{Question:} Write an ALP to read an alphanumeric character and display its ASCII code at the center of the screen.
\end{minipage}%
}
\vspace{0.4em}
\noindent\textbf{Answer}
\begin{lstlisting}[style=asmbase]
DATA SEGMENT
MSG1 DB 0DH,0AH,'ENTER A CHARACTER:$'
MSG2 DB 0DH,0AH,'ASCII CODE: $'
DATA ENDS

CODE SEGMENT
ASSUME CS:CODE,DS:DATA
START:
    MOV AX,DATA
    MOV DS,AX

    MOV AH,09H
    LEA DX,MSG1
    INT 21H

    MOV AH,01H
    INT 21H
    MOV BL,AL

    MOV AH,02H
    MOV BH,00H
    MOV DH,12
    MOV DL,38
    INT 10H

    MOV AH,09H
    LEA DX,MSG2
    INT 21H

    MOV AL,BL
    MOV CL,04
    MOV DL,AL
    SHR DL,CL
    CALL HEX_DIGIT

    MOV DL,BL
    AND DL,0FH
    CALL HEX_DIGIT

    MOV AH,4CH
    INT 21H

HEX_DIGIT PROC NEAR
    CMP DL,09H
    JBE DIG
    ADD DL,07H
DIG:
    ADD DL,30H
    MOV AH,02H
    INT 21H
    RET
HEX_DIGIT ENDP

CODE ENDS
END START
\end{lstlisting}

\section*{Experiment 4}
\addcontentsline{toc}{section}{Experiment 4}
\noindent\fcolorbox{linegray}{qbg}{%
\begin{minipage}{0.97\linewidth}
\textbf{Question:} Write an ALP to reverse a given string and check whether it is palindrome or not.
\end{minipage}%
}
\vspace{0.4em}
\noindent\textbf{Answer}
\begin{lstlisting}[style=asmbase]
DATA SEGMENT
STR1 DB 50 DUP(?)
STR2 DB 50 DUP(?)
N DB ?
M1 DB 0DH,0AH,'ENTER A STRING:$'
M2 DB 0DH,0AH,'IT IS PALINDROME$'
M3 DB 0DH,0AH,'IT IS NOT PALINDROME$'
DATA ENDS

CODE SEGMENT
ASSUME CS:CODE,DS:DATA
START:
    MOV AX,DATA
    MOV DS,AX

    MOV AH,09H
    LEA DX,M1
    INT 21H

    LEA SI,STR1
    MOV CL,00
READS:
    MOV AH,01H
    INT 21H
    CMP AL,0DH
    JE DONE_READ
    MOV [SI],AL
    INC SI
    INC CL
    JMP READS

DONE_READ:
    MOV N,CL

    DEC SI
    LEA DI,STR2
    MOV CH,00
REV:
    CMP CH,N
    JAE CHK
    MOV AL,[SI]
    MOV [DI],AL
    DEC SI
    INC DI
    INC CH
    JMP REV

CHK:
    LEA SI,STR1
    LEA DI,STR2
    MOV CL,00
CMP_LOOP:
    CMP CL,N
    JAE PAL
    MOV AL,[SI]
    CMP AL,[DI]
    JNE NOTPAL
    INC SI
    INC DI
    INC CL
    JMP CMP_LOOP

PAL:
    MOV AH,09H
    LEA DX,M2
    INT 21H
    JMP EXIT

NOTPAL:
    MOV AH,09H
    LEA DX,M3
    INT 21H

EXIT:
    MOV AH,4CH
    INT 21H
CODE ENDS
END START
\end{lstlisting}

\section*{Experiment 5}
\addcontentsline{toc}{section}{Experiment 5}
\noindent\fcolorbox{linegray}{qbg}{%
\begin{minipage}{0.97\linewidth}
\textbf{Question:} Write an ALP to compute factorial of a positive integer n using recursive procedure.
\end{minipage}%
}
\vspace{0.4em}
\noindent\textbf{Answer}
\begin{lstlisting}[style=asmbase]
DATA SEGMENT
NUM DB 04H
FACT DW ?
MSG DB 0DH,0AH,'FACTORIAL: $'
DATA ENDS

CODE SEGMENT
ASSUME CS:CODE,DS:DATA
START:
    MOV AX,DATA
    MOV DS,AX

    MOV BL,NUM
    XOR BH,BH
    MOV AX,0001H
    CALL FACTORIAL
    MOV FACT,AX

    MOV AH,09H
    LEA DX,MSG
    INT 21H

    MOV BX,FACT
    CALL DISPHEX

    MOV AH,4CH
    INT 21H

FACTORIAL PROC NEAR
    CMP BX,0001H
    JBE FRET
    PUSH BX
    DEC BX
    CALL FACTORIAL
    POP BX
    MUL BX
FRET:
    RET
FACTORIAL ENDP

DISPHEX PROC NEAR
    PUSH CX
    PUSH DX
    MOV CX,4
HX1:
    ROL BX,1
    ROL BX,1
    ROL BX,1
    ROL BX,1
    MOV DL,BL
    AND DL,0FH
    CMP DL,09H
    JBE HX2
    ADD DL,07H
HX2:
    ADD DL,30H
    MOV AH,02H
    INT 21H
    LOOP HX1
    POP DX
    POP CX
    RET
DISPHEX ENDP

CODE ENDS
END START
\end{lstlisting}

\section*{Experiment 6}
\addcontentsline{toc}{section}{Experiment 6}
\noindent\fcolorbox{linegray}{qbg}{%
\begin{minipage}{0.97\linewidth}
\textbf{Question:} Write an ALP to generate the first n Fibonacci numbers.
\end{minipage}%
}
\vspace{0.4em}
\noindent\textbf{Answer}
\begin{lstlisting}[style=asmbase]
DATA SEGMENT
F1 DB 00H
F2 DB 01H
F3 DB ?
N DB 0AH
MSG DB 0DH,0AH,'FIBONACCI SERIES: $'
DATA ENDS

CODE SEGMENT
ASSUME CS:CODE,DS:DATA
START:
    MOV AX,DATA
    MOV DS,AX

    MOV AH,09H
    LEA DX,MSG
    INT 21H

    MOV BL,F1
    CALL DISPHEX2
    MOV DL,' '
    MOV AH,02H
    INT 21H

    MOV BL,F2
    CALL DISPHEX2
    MOV DL,' '
    MOV AH,02H
    INT 21H

    MOV CL,2
LOOPF:
    CMP CL,N
    JAE EXIT

    MOV AL,F1
    ADD AL,F2
    MOV F3,AL

    MOV BL,F3
    CALL DISPHEX2
    MOV DL,' '
    MOV AH,02H
    INT 21H

    MOV AL,F2
    MOV F1,AL
    MOV AL,F3
    MOV F2,AL

    INC CL
    JMP LOOPF

EXIT:
    MOV AH,4CH
    INT 21H

DISPHEX2 PROC NEAR
    PUSH AX
    PUSH CX
    PUSH DX
    MOV DL,BL
    MOV CL,04
    SHR DL,CL
    CALL HEXD
    MOV DL,BL
    AND DL,0FH
    CALL HEXD
    POP DX
    POP CX
    POP AX
    RET
DISPHEX2 ENDP

HEXD PROC NEAR
    CMP DL,09H
    JBE D1
    ADD DL,07H
D1:
    ADD DL,30H
    MOV AH,02H
    INT 21H
    RET
HEXD ENDP

CODE ENDS
END START
\end{lstlisting}

DISP2 ENDP

\section*{Experiment 8}
\addcontentsline{toc}{section}{Experiment 8}
\noindent\fcolorbox{linegray}{qbg}{%
\begin{minipage}{0.97\linewidth}
\textbf{Question:} Write an ALP to create/open myfile.txt, write text, close, reopen, read and display file content.
\end{minipage}%
}
\vspace{0.4em}
\noindent\textbf{Answer}
\begin{lstlisting}[style=asmbase]
DATA SEGMENT
FILE DB 'MYFILE.TXT',0
MSG DB 'WELCOME TO MICROPROCESSOR LAB'
MSGLEN EQU ($-MSG)
BUF DB 64 DUP(?)
READCNT DW ?
DATA ENDS

CODE SEGMENT
ASSUME CS:CODE,DS:DATA
START:
    MOV AX,DATA
    MOV DS,AX

    LEA DX,FILE
    MOV CX,00H
    MOV AH,3CH
    INT 21H
    MOV BX,AX

    LEA DX,MSG
    MOV CX,MSGLEN
    MOV AH,40H
    INT 21H

    MOV AH,3EH
    INT 21H

    LEA DX,FILE
    MOV AL,00H
    MOV AH,3DH
    INT 21H
    MOV BX,AX

    LEA DX,BUF
    MOV CX,64
    MOV AH,3FH
    INT 21H
    MOV READCNT,AX

    MOV AH,3EH
    INT 21H

    MOV BX,1
    LEA DX,BUF
    MOV CX,READCNT
    MOV AH,40H
    INT 21H

    MOV AH,4CH
    INT 21H
CODE ENDS
END START
\end{lstlisting}
