package invoiceTest;

class invoice {
    int itemNo;
    String itemName;
    int quantity;
    int perItem;

    int getItemNo() {
        return itemNo;
    }

    String getItemName() {
        return itemName;
    }

    int getQuantity() {
        return quantity;
    }

    int getPerItem() {
        return perItem;
    }

    void setItemNo(int itemNo) {
        this.itemNo = itemNo;
    }

    void setItemName(String itemName) {
        this.itemName = itemName;
    }

    void setQuantity(int quantity) {
        this.quantity = quantity;
    }

    void setPerItem(int perItem) {
        this.perItem = perItem;
    }

    double getAmount() {
        if (quantity <= 0 || perItem <= 0) {
            quantity = 0;
            perItem = 0;
        }
        double result = quantity * perItem;
        return result;
    }
}

public class InvoiceTest {

    public static void main(String[] args) {
        invoice i = new invoice();
        i.setItemName("Book");
        i.setItemNo(1);
        i.setQuantity(1);
        i.setPerItem(89);

        System.out.println("Item Name: " + i.getItemName());
        System.out.println("Item Number: " + i.getItemNo());
        System.out.println("Quantity: " + i.getQuantity());
        System.out.println("Price Per Item: " + i.getPerItem());
        System.out.println("Total Amount: " + i.getAmount());

    }
}
